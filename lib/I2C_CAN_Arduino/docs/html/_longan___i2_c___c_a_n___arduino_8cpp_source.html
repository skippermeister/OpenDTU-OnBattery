245unsigned char I2C_CAN::makeCheckSum(unsigned char *dta, int len)
246{
247 unsigned long sum = 0;
248 for(int i=0; i<len; i++)sum += dta[i];
249
250 if(sum > 0xff)

252 sum = ~sum;

254 }
