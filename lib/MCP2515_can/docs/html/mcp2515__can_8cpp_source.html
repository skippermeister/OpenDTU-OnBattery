63uint8_t MCP2515Class::initCANBuffers(void)
64{
65 // clear filters and masks
66 // do not filter any standard frames for RXF0 used by RXB0
67 // do not filter any extended frames for RXF1 used by RXB1
68 const RXF_t filters[] = {RXF0, RXF1, RXF2, RXF3, RXF4, RXF5};
69 for (int i=0; i<6; i++) {
70 const bool ext = (i == 1);
71 int rc = setFilter(filters[i], ext, 0);
72 if (rc != CAN_OK) {
73 return rc;
74 }
75 }
76
77 MASK_t masks[] = {MASK0, MASK1};
78 for (int i=0; i<2; i++) {
79 int rc = setFilterMask(masks[i], true, 0);
80 if (rc != CAN_OK) {
81 return rc;
82 }
83 }
84
85 /* Clear, deactivate the three transmit buffers
86 * TXBnCTRL -> TXBnD7
87 */
88 uint8_t zeros[14];
89 memset(zeros, 0, sizeof(zeros));
90 spi_setRegisters(MCP_TXB0CTRL, zeros, 14);
91 spi_setRegisters(MCP_TXB1CTRL, zeros, 14);
92 spi_setRegisters(MCP_TXB2CTRL, zeros, 14);
93
94 spi_setRegister(MCP_RXB0CTRL, 0);
95 spi_setRegister(MCP_RXB1CTRL, 0);
96
97 return CAN_OK;
98}

100uint8_t MCP2515Class::reset(const int8_t canIDMode, const CAN_SPEED_t canSpeed, const MCP2515_CLOCK_t canClock)
101{
102 spi_reset();
103
104 CANCTRL_REQOP_MODE_t mcpMode = MCP_LOOPBACK;
105
106 if (setMode(MODE_CONFIG) != CAN_OK)
107 {
108 Serial.println(F("Entering Configuration Mode Failure..."));
109 return CAN_FAILINIT;
110 }
111 Serial.println(F("Entering Configuration Mode Successful!"));
112
113 vTaskDelay(10 / portTICK_PERIOD_MS);
114 // Set Baudrate
115 if(setBitrate(canSpeed, canClock) != CAN_OK)
116 {
117 Serial.println("Setting Baudrate Failure...");
118 return CAN_FAILINIT;
119 }
120 Serial.println("Setting Baudrate Successful!");
121
122 if (initCANBuffers() != CAN_OK) {
123 Serial.println(F("Initializing CAN buffers Failure..."));
124 return CAN_FAILINIT;
125 }
126
127 // spi_setRegister(MCP_CANINTE, CANINTF_RX0IF | CANINTF_RX1IF | CANINTF_ERRIF | CANINTF_MERRF);
128 spi_setRegister(MCP_CANINTE, CANINTF_RX0IF | CANINTF_RX1IF);
129
130 //Sets BF pins as GPO
131 spi_setRegister(MCP_BFPCTRL, MCP_BxBFS_MASK | MCP_BxBFE_MASK);
132 //Sets RTS pins as GPI
133 spi_setRegister(MCP_TXRTSCTRL, 0x00);
134
135 switch(canIDMode)
136 {
137 case (MCP_ANY):
138 spi_modifyRegister(MCP_RXB0CTRL,
139 RXBnCTRL_RXM_MASK | RXB0CTRL_BUKT,
140 RXBnCTRL_RXM_ANY | RXB0CTRL_BUKT);
141 spi_modifyRegister(MCP_RXB1CTRL, RXBnCTRL_RXM_MASK, RXBnCTRL_RXM_ANY);
142 break;
143
144 case (MCP_STDEXT):
145 // receives all valid messages using either Standard or Extended Identifiers that
146 // meet filter criteria. RXF0 is applied for RXB0, RXF1 is applied for RXB1
147 spi_modifyRegister(MCP_RXB0CTRL,
148 RXBnCTRL_RXM_MASK | RXB0CTRL_BUKT | RXB0CTRL_FILHIT_MASK,
149 RXBnCTRL_RXM_STDEXT | RXB0CTRL_BUKT | RXB0CTRL_FILHIT);
150 spi_modifyRegister(MCP_RXB1CTRL,
151 RXBnCTRL_RXM_MASK | RXB1CTRL_FILHIT_MASK,
152 RXBnCTRL_RXM_STDEXT | RXB1CTRL_FILHIT);
153 break;
154
155 default:
156 Serial.println("Setting ID Mode Failure...");
157 return CAN_FAILINIT;
158 }
159
160 if (setMode(mcpMode) != CAN_OK)
161 {
162 Serial.println(F("Returning to Previous Mode Failure..."));
163 return CAN_FAILINIT;
164 }
165
166 return CAN_OK;
167}

174uint8_t MCP2515Class::setOneShotMode(bool set)
175{
176 uint8_t data = 0;
177 if (set)
178 data = MODE_ONESHOT;
179 spi_modifyRegister(MCP_CANCTRL, MODE_ONESHOT, data);

182 for (int i = 0; i < 10; i++) {
183 uint8_t ctrlR = spi_readRegister(MCP_CANCTRL);

185 if (modeMatch)
186 break;
187 vTaskDelay(10 / portTICK_PERIOD_MS);
188 }

234uint8_t MCP2515Class::setMode(const CANCTRL_REQOP_MODE_t mode)
235{
236 // If the chip is asleep and we want to change mode then a manual wake needs to be done
237 // This is done by setting the wake up interrupt flag
238 // This undocumented trick was found at https://github.com/mkleemann/can/blob/master/can_sleep_mcp2515.c
239 if((spi_readRegister(MCP_CANSTAT) & MODE_MASK) == MCP_SLEEP && mode != MCP_SLEEP)
240 {
241 // Make sure wake interrupt is enabled
242 byte wakeIntEnabled = (spi_readRegister(MCP_CANINTE) & CANINTF_WAKIF);
243 if(!wakeIntEnabled)
244 spi_modifyRegister(MCP_CANINTE, CANINTF_WAKIF, CANINTF_WAKIF);
245
246 // Set wake flag (this does the actual waking up)
247 spi_modifyRegister(MCP_CANINTF, CANINTF_WAKIF, CANINTF_WAKIF);
248
249 // Wait for the chip to exit SLEEP and enter LISTENONLY mode.
250
251 // If the chip is not connected to a CAN bus (or the bus has no other powered nodes) it will sometimes trigger the wake interrupt as soon
252 // as it's put to sleep, but it will stay in SLEEP mode instead of automatically switching to LISTENONLY mode.
253 // In this situation the mode needs to be manually set to LISTENONLY.
254
255 if(requestNewMode(MCP_LISTENONLY) != CAN_OK)
256 return CAN_FAIL;
257
258 // Turn wake interrupt back off if it was originally off
259 if(!wakeIntEnabled)
260 spi_modifyRegister(MCP_CANINTE, CANINTF_WAKIF, 0);
261
262 }
263
264
265 // Clear wake flag
266 spi_modifyRegister(MCP_CANINTF, CANINTF_WAKIF, 0);
267
268 return requestNewMode(mode);
269}

271/*********************************************************************************************************
272** Function name: requestNewMode
273** Descriptions: Set control mode
274*********************************************************************************************************/
275uint8_t MCP2515Class::requestNewMode(const uint8_t newmode)
276{
277 byte startTime = millis();
278
279 // Spam new mode request and wait for the operation to complete
280 while(1)
281 {
282 // Request new mode
283 // This is inside the loop as sometimes requesting the new mode once doesn't work (usually when attempting to sleep)
284 spi_modifyRegister(MCP_CANCTRL, CANSTAT_OPMOD, newmode);
285
286 byte statReg = spi_readRegister(MCP_CANSTAT);
287 if((statReg & MODE_MASK) == newmode) // We're now in the new mode
288 return CAN_OK;
289 else if((byte)(millis() - startTime) > 200) // Wait no more than 200ms for the operation to complete
290 return CAN_FAIL;
291 }
292}

604void MCP2515Class::prepareId(uint8_t *buffer, const bool ext, const uint32_t id)
605{
606 uint16_t canid = (uint16_t)(id & 0x0FFFF);
607
608 if (ext) {
609 buffer[MCP_EID0] = (uint8_t) (canid & 0xFF);
610 buffer[MCP_EID8] = (uint8_t) (canid >> 8);
611 canid = (uint16_t)(id >> 16);
612 buffer[MCP_SIDL] = (uint8_t) (canid & 0x03);
613 buffer[MCP_SIDL] += (uint8_t) ((canid & 0x1C) << 3);
614 buffer[MCP_SIDL] |= TXB_EXIDE_MASK;
615 buffer[MCP_SIDH] = (uint8_t) (canid >> 5);
616 } else {
617 buffer[MCP_SIDH] = (uint8_t) (canid >> 3);
618 buffer[MCP_SIDL] = (uint8_t) ((canid & 0x07 ) << 5);
619 buffer[MCP_EID0] = 0;
620 buffer[MCP_EID8] = 0;
621 }

738uint8_t MCP2515Class::readMessage(const RXBn_t rxbn, can_message_t *rx_message)
739{
740 uint8_t rc = CAN_OK;
741 uint8_t tbufdata[5];
742 const auto rxb = &RXB_ptr[rxbn];
743
744 rx_message->identifier = 0;
745 rx_message->data_length_code = 0;
746 rx_message->flags = 0;
747 memset(rx_message->data, 0, sizeof(rx_message->data));
748
749 spi_readRegisters(rxb->SIDH, tbufdata, 5); // read 4 bytes of id and dlc
750
751 uint32_t id = (tbufdata[MCP_SIDH]<<3) + (tbufdata[MCP_SIDL]>>5);
752
753 if ( (tbufdata[MCP_SIDL] & TXB_EXIDE_MASK) == TXB_EXIDE_MASK ) {
754 // extended id
755 id = (id<<2) + (tbufdata[MCP_SIDL] & 0x03);
756 id = (id<<8) + tbufdata[MCP_EID8];
757 id = (id<<8) + tbufdata[MCP_EID0];
758 rx_message->extd = 1;
759 }
760
761 uint8_t dlc = (tbufdata[MCP_DLC] & DLC_MASK);
762 if (dlc > CAN_MAX_DLC) {
763 rx_message->dlc_non_comp = 1;
764 dlc = CAN_MAX_DLC;
765 rc = CAN_FAILRX_DLC;
766 }
767 if (spi_readRegister(rxb->CTRL) & RXBnCTRL_RTR) {
768 rx_message->rtr = 1;
769 }
770
771 rx_message->identifier = id;
772 rx_message->data_length_code = dlc;
773
774 spi_readRegisters(rxb->DATA, rx_message->data, dlc);
775
776 spi_modifyRegister(MCP_CANINTF, rxb->CANINTF_RXnIF, 0);
777
778 return rc;
779}
